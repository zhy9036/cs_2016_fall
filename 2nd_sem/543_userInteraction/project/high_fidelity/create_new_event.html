function checkDate() {

     var date_regex = /^(0[1-9]|1[012])[- \/.](0[1-9]|[12][0-9]|3[01])[- \/.]((19|20)\d\d)$/;

     var endDate= document.getElementById("u41_input");

     var endTime= document.getElementById("u42_input");

     var startDate= document.getElementById("u35_input");

     var startTime = document.getElementById("u37_input");

     var eventName = document.getElementById("u34_input");


     var time_reg = /^(0[1-9]|1[012])[:](0[1-9]|[0-5][0-9])$/;

     if(eventName.value == ""){

          alert("Event Name cannot be empty");

     }else if(!startDate.value.match(date_regex) || !endDate.value.match(date_regex)){

          alert("Date format not corrent! Should be in mm/dd/yyyy");

    

     }else if(!startTime.value.match(time_reg) || !endTime.value.match(time_reg)){

          alert("Time format not corrent! Should be in hh/mm");

     }else{

          alert("Event created!");

          location.href = 'home_1.html';

     }

}
